281 template <typename T>
282 explicit operator T() const {
283 static_assert(sizeof(T) <= 4 || type == Type::String, "Target type is too large");
284 switch (type) {
285 case Type::Invalid:
286 return static_cast<T>(NULL);
287 case Type::Float:
288 return static_cast<T>(f);
289 case Type::Uint:
290 return static_cast<T>(u);
291 case Type::Int:
292 return static_cast<T>(i);
293 case Type::Bool:
294 return static_cast<T>(b);
295 case Type::String:
296 if constexpr (std::is_same_v<T, float>)
297 return s ? static_cast<T>(std::atof(s)) : 0.0f;
298 else if constexpr (std::is_same_v<T, int32_t> || std::is_same_v<T, uint32_t>)
299 return s ? static_cast<T>(std::atoi(s)) : 0;
300 else if constexpr (std::is_same_v<T, bool>)
301 return s != nullptr && s[0] != '\0';
302 else
303 return T{};
304 }
305
306 std::unreachable();
307 }
