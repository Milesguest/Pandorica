299 template <typename T>
300 explicit operator T() const {
301 static_assert(sizeof(T) <= 4 || type == Type::String, "Target type is too large");
302 switch (type) {
303 case Type::Invalid:
304 return static_cast<T>(NULL);
305 case Type::Float:
306 return static_cast<T>(f);
307 case Type::Uint:
308 return static_cast<T>(u);
309 case Type::Int:
310 return static_cast<T>(i);
311 case Type::Bool:
312 return static_cast<T>(b);
313 case Type::String:
314 if constexpr (std::is_same_v<T, float>)
315 return s ? static_cast<T>(std::atof(s)) : 0.0f;
316 else if constexpr (std::is_same_v<T, int32_t> || std::is_same_v<T, uint32_t>)
317 return s ? static_cast<T>(std::atoi(s)) : 0;
318 else if constexpr (std::is_same_v<T, bool>)
319 return s != nullptr && s[0] != '\0';
320 else
321 return T{};
322 }
323
324 std::unreachable();
325 }
