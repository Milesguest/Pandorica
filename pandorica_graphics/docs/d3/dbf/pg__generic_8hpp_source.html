353 template <typename T>
354 explicit operator T() const {
355 static_assert(sizeof(T) <= 4 || type == Type::String, "Target type is too large");
356 switch (type) {
357 case Type::Invalid:
358 return static_cast<T>(NULL);
359 case Type::Float:
360 return static_cast<T>(f);
361 case Type::Uint:
362 return static_cast<T>(u);
363 case Type::Int:
364 return static_cast<T>(i);
365 case Type::Bool:
366 return static_cast<T>(b);
367 case Type::String:
368 if constexpr (std::is_same_v<T, float>)
369 return s ? static_cast<T>(std::atof(s)) : 0.0f;
370 else if constexpr (std::is_same_v<T, int32_t> || std::is_same_v<T, uint32_t>)
371 return s ? static_cast<T>(std::atoi(s)) : 0;
372 else if constexpr (std::is_same_v<T, bool>)
373 return s != nullptr && s[0] != '\0';
374 else
375 return T{};
376 }
377
378 std::unreachable();
379 }
